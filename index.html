2. Le code JavaScript (Logique du Générateur) Ajoute ou remplace ces fonctions dans ton script.js. J'ai ajouté la logique pour récupérer les chapitres de la base Supabase et lancer le mode chrono. JavaScript // --- VARIABLES GLOBALES QUIZ --- let currentQuizQuestions = []; let quizTimerIdx = null; let quizScore = 0; let timeLeft = 60; // 1. Préparer le générateur (Charger les chapitres de la matière actuelle) async function prepareQuizGenerator() { const container = document.getElementById('quiz-chapter-selection'); container.innerHTML = "

Chargement des chapitres...

"; navigateTo('view-quiz-generator'); // Récupérer les chapitres uniques depuis Supabase pour cette matière const { data, error } = await sb .from('lessons') .select('chapter_number, content') .eq('class_id', state.currentClassCode) .eq('subject_id', state.currentSubject.toLowerCase()); if (error || !data) return alert("Erreur chargement chapitres"); container.innerHTML = ""; data.forEach(item => { // Extraction du titre (H1) du contenu const temp = document.createElement('div'); temp.innerHTML = item.content; const title = temp.querySelector('h1')?.innerText || "Chapitre " + item.chapter_number; const div = document.createElement('div'); div.className = "chapter-item"; div.dataset.chapId = item.chapter_number; div.innerText = title; div.onclick = () => div.classList.toggle('selected'); container.appendChild(div); }); } // 2. Lancer le quiz personnalisé async function startGeneratedQuiz() { const selected = Array.from(document.querySelectorAll('.chapter-item.selected')).map(el => el.dataset.chapId); if (selected.length === 0) return alert("Sélectionne au moins un chapitre !"); // Récupérer toutes les questions des chapitres sélectionnés const { data, error } = await sb .from('es') .select('*') .eq('class_id', state.currentClassCode) .eq('subject_id', state.currentSubject.toLowerCase()) .in('chapter_number', selected); if (error || !data || data.length === 0) return alert("Aucune question trouvée pour ces chapitres."); currentQuizQuestions = data.sort(() => 0.5 - Math.random()); quizScore = 0; document.getElementById('quiz-score').innerText = "Score: 0"; // Switch affichage document.querySelector('.quiz-setup').style.display = 'none'; document.getElementById('quiz-active-area').style.display = 'block'; // Mode Chrono ? const isTimer = document.getElementById('timer-mode').checked; if (isTimer) { startQuizCountdown(); } else { document.getElementById('quiz-timer').innerText = "Mode Libre"; } renderNextQuestion(0); } // 3. Gestion du Timer function startQuizCountdown() { timeLeft = 60; const timerDisplay = document.getElementById('quiz-timer'); timerDisplay.innerText = "01:00"; if(quizTimerIdx) clearInterval(quizTimerIdx); quizTimerIdx = setInterval(() => { timeLeft--; timerDisplay.innerText = `00:${timeLeft < 10 ? '0' : ''}${timeLeft}`; if (timeLeft <= 0) { clearInterval(quizTimerIdx); endQuizChallenge(); } }, 1000); } // 4. Affichage des questions function renderNextQuestion(idx) { const container = document.getElementById('quiz-question-container'); if (idx >= currentQuizQuestions.length) return endQuizChallenge(); const q = currentQuizQuestions[idx]; container.innerHTML = `
${q.question}
`; const optionsDiv = container.querySelector('#options-list'); q.options.forEach((opt, optIdx) => { const btn = document.createElement('div'); btn.className = "quiz-option"; btn.style = "padding:1rem; border:2px solid #eee; margin-bottom:10px; border-radius:10px; cursor:pointer;"; btn.innerText = opt; btn.onclick = () => { if (optIdx === q.correct_index) { quizScore++; document.getElementById('quiz-score').innerText = "Score: " + quizScore; renderNextQuestion(idx + 1); // Question suivante direct pour la vitesse } else { btn.style.borderColor = "var(--accent-red)"; btn.style.background = "#fee"; // Petite pause avant de continuer si erreur setTimeout(() => renderNextQuestion(idx + 1), 500); } }; optionsDiv.appendChild(btn); }); } function endQuizChallenge() { clearInterval(quizTimerIdx); alert("Défi terminé ! Ton score final est de " + quizScore); // Reset pour le prochain coup document.querySelector('.quiz-setup').style.display = 'block'; document.getElementById('quiz-active-area').style.display = 'none'; navigateTo('view-home'); }
